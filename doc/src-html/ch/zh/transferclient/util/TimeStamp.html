017package ch.zh.transferclient.util;
018
019import java.text.SimpleDateFormat;
020import java.util.Date;
021
022/**
023 * This class is used to generate TimeStamps.
024 *
025 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
026 * @version 2.4
027 */
028public class TimeStamp
029    
030    {
031    
032    /** Date format for the sedex message id. */
033    private static final SimpleDateFormat SDF_FOR_SEDEX_MESSAGE_ID = new SimpleDateFormat("yyyyMMdd-HHmm-ss-SSS");
034    /** Date format for the sedex envelope. */
035    private static final SimpleDateFormat SDF_FOR_SEDEX_ENVELOPE   = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
036    /** Date format for the log file. */
037    // private static final SimpleDateFormat SDF_FOR_LOGFILE = new
038    // SimpleDateFormat("yyyy_MM_dd_HH_mm_ss");
039    private static final SimpleDateFormat SDF_FOR_LOGFILE          = new SimpleDateFormat("yyyyMMdd-HHmm-ss-SSS");
040    
041    /**
042     * Constructs a TimeStamp object.
043     */
044    private TimeStamp()
045        {
046        //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
047        }
048    
049    /**
050     * Returns the date format for the sedex message id.
051     * 
052     * @return Date format for the sedex message id.
053     */
054    public static String getstamp_for_sedex_message_id()
055        {
056        return SDF_FOR_SEDEX_MESSAGE_ID.format(new Date());
057        }
058        
059    /**
060     * Returns the date format for the sedex envelope.
061     * 
062     * @return Date format for the sedex envelope.
063     */
064    public static String getstamp_for_sedex_envelope()
065        {
066        return SDF_FOR_SEDEX_ENVELOPE.format(new Date());
067        }
068        
069    /**
070     * Returns the date format for the log file.
071     * 
072     * @return Date format for the log file.
073     */
074    public static String getstamp_for_logfile()
075        {
076        return SDF_FOR_LOGFILE.format(new Date());
077        }
078        
079    }
