017package ch.zh.transferclient.main;
018
019import java.io.File;
020import java.nio.file.Files;
021
022import javax.swing.SwingUtilities;
023
024import static java.nio.file.StandardCopyOption.*;
025
026import ch.zh.transferclient.gui.*;
027import ch.zh.transferclient.properties.*;
028
029
030/**
031 * This class is used to delete old receipts.
032 *
033 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
034 * @version 2.4
035 */
036public class Archiving_Old_Receipts
037    
038    {
039    
040    /**
041     * Constructs a Archiving_Old_Receipts object.
042     */
043    private Archiving_Old_Receipts()
044        {
045      //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
046        }
047    
048    /**
049     * Deletes all files within the sedex receipts folder.
050     *
051     * @param  properties Properties to be used.
052     * @param  guistart   The guistart object to be used.
053     * @throws Exception  The treatment of old properties failed.
054     */
055    protected static void execute(Properties properties,GuiStart guistart) throws Exception
056        {
057        
058        String       sedex_dir_receipts = properties.get_sedex_dir_receipts();
059        
060        final File   dir                = new File(sedex_dir_receipts);
061        
062        final File[] files              = dir.listFiles();
063        
064        if (files.length==0)
065            {
066            SwingUtilities.invokeAndWait(new Runnable()
067                {
068                @Override
069                public void run()
070                    {
071                    try
072                        {
073                        String update_text = "0 of 0 old receipts archived.";
074                        guistart.update_statusfield(GuiStart.StatusField.STATUSFIELD1,update_text);
075                        }
076                    catch(Exception e)
077                        {
078                        // Hier koennte hoechstens eine null pointer exception geworfen werden,
079                        // wenn das guistart-Objekt noch nicht durch den EDT-Thread fertig 
080                        // konstruiert ist (vgl. main-Methode). Durch Verwendung von invokeAndWait 
081                        // in der main-Methode ist jedoch sichergestellt, dass das 
082                        // guistart-Objekt bereits durch den EDT-Thread erstellt worden ist. 
083                        
084                        // Damit die null pointer exception auch geworfen wird, muss sich
085                        // der try-catch-Block innerhalb des Runnable Objekts befinden,
086                        // da sonst nur eine nichtssagende  
087                        // java.lang.reflect.InvocationTargetException Ausnahme geworfen
088                        // werden wuerde.
089                        }
090                    }
091                });
092            }
093        
094       
095        for (int i = 0; i < files.length; i++)
096            {
097            
098            Files.move(files[i].toPath(), new File("archive/receipts/"
099                    + files[i].getName()).toPath(), REPLACE_EXISTING);
100            
101            final int i_final = i;
102            
103            SwingUtilities.invokeAndWait(new Runnable()
104                {
105                @Override
106                public void run()
107                    {
108                    try
109                        {
110                        String update_text = i_final + " of " + files.length + " old receipts archived.";
111                        guistart.update_statusfield(GuiStart.StatusField.STATUSFIELD1,update_text);
112                        }
113                    catch(Exception e)
114                        {
115                        // Hier koennte hoechstens eine null pointer exception geworfen werden,
116                        // wenn das guistart-Objekt noch nicht durch den EDT-Thread fertig 
117                        // konstruiert ist (vgl. main-Methode). Durch Verwendung von invokeAndWait 
118                        // in der main-Methode ist jedoch sichergestellt, dass das 
119                        // guistart-Objekt bereits durch den EDT-Thread erstellt worden ist. 
120                        
121                        // Damit die null pointer exception auch geworfen wird, muss sich
122                        // der try-catch-Block innerhalb des Runnable Objekts befinden,
123                        // da sonst nur eine nichtssagende  
124                        // java.lang.reflect.InvocationTargetException Ausnahme geworfen
125                        // werden wuerde.
126                        }
127                    
128                    }
129                });
130            
131            
132            Logger.info("OLD RECEIPT MOVED TO archive/receipts: " + files[i].getName());
133            
134            }
135        
136        //Nachfolgend werden Files ins Archiv verschoben, 
137        //die aufgrund von ausserordentlichen Unterbruechen 
138        //im Stage-Folder zurueckgeblieben sind.
139        File dir_stage = new File("stage/");
140        final File[] files_stage = dir_stage.listFiles();
141        
142        for (int i = 0; i < files_stage.length; i++)
143            {
144            
145            Files.move(files_stage[i].toPath(), new File("archive/stage/"
146                    + files_stage[i].getName()).toPath(), REPLACE_EXISTING);
147            
148            Logger.info("UNPROCESSED FILE MOVED FROM stage TO archive/stage: " + files[i].getName());
149            
150            }
151            
152        }
153        
154    }
