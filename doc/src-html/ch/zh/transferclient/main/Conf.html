017package ch.zh.transferclient.main;
018
019import java.awt.*;
020import java.net.URL;
021import java.util.*;
022
023import javax.swing.ImageIcon;
024
025/**
026 * This class is used to specify parameters which are not accessible by the external configuration file.
027 *
028 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
029 * @version 2.4
030 */
031public class Conf
032    
033    {
034    
035    /** Version of the Transfer-Client. */
036    public static final String            VERSION                     = "2.4";
037    
038    /**
039     * Indicator which indicates whether the log should be written also to the console.
040     */
041    public static final boolean           LOG_ALSO_TO_CONSOLE         = false;
042    
043    
044    // GUI
045    
046    /** Icon: URL of the icon. */
047    public static final URL               IMAGE_URL                   = ClassLoader.getSystemResource("ch/zh/transferclient/gui/icon.jpg");
048    
049    /** Icon: ImageIcon object. */
050    public static final ImageIcon         IMAGE_ICON                  = new ImageIcon(IMAGE_URL);
051    
052    /** Icon: Image object. */
053    public static final Image             IMAGE_FOR_ICON              = IMAGE_ICON.getImage();
054    
055    /** Logo: URL of the logo. */
056    public static final URL               IMAGE_URL_LOGO              = ClassLoader.getSystemResource("ch/zh/transferclient/gui/logo.jpg");
057    
058    /** Logo: ImagIcon object. */
059    public static final ImageIcon         IMAGE_ICON_LOGO             = new ImageIcon(IMAGE_URL_LOGO);
060    
061    /** GREEN-Color object. */
062    public static final Color             GREEN                       = new Color(0, 153, 51);
063    
064    /** RED-Color object. */
065    public static final Color             RED                         = Color.RED;
066    
067    /**
068     * Indicator which indicates whether the process of deactivation should be signaled by a dialog (only under the
069     * deprecated NIO_WATCHSERVICE approach).
070     */
071    public static final boolean           DEACTIVATION_WITH_DIALOG    = true;
072    
073    // Zeitauswahl bei Autoaktivierung
074    /** Configuration parameter of the auto-activation menu: start hour. */
075    private static final int              AUTOACTIVATION_HOUR_START   = 9;                                                                 // Default-Wert:
076                                                                                                                                           // 9
077    /** Configuration parameter of the auto-activation menu: end hour. */
078    private static final int              AUTOACTIVATION_HOUR_END     = 13;                                                                // Default-Wert:
079                                                                                                                                           // 13
080    /** Configuration parameter of the auto-activation menu: start minute. */
081    private static final int              AUTOACTIVATION_MINUTE_START = 0;                                                                 // Default-Wert:
082                                                                                                                                           // 0
083    /** Configuration parameter of the auto-activation menu: end minute. */
084    private static final int              AUTOACTIVATION_MINUTE_END   = 55;                                                                // Default-Wert:
085                                                                                                                                           // 55
086    /**
087     * Configuration parameter of the auto-activation menu: frequency in minutes.
088     */
089    private static final int              AUTOACTIVATION_MINUTE_FREQ  = 5;                                                                 // Default-Wert:
090                                                                                                                                           // 5
091    /** Configuration parameter of the auto-activation menu: points of time. */
092    public static final ArrayList<String> AUTOACTIVATION_TIMES        = new ArrayList<String>();
093    
094    static
095        {
096        
097        for (int i = AUTOACTIVATION_HOUR_START; i <= AUTOACTIVATION_HOUR_END; i++)
098            {
099            
100            if (i == AUTOACTIVATION_HOUR_END)
101                {
102                AUTOACTIVATION_TIMES.add(i + ":00");
103                }
104            else
105                {
106                
107                for (int j = AUTOACTIVATION_MINUTE_START; j <= AUTOACTIVATION_MINUTE_END; j = j
108                        + AUTOACTIVATION_MINUTE_FREQ)
109                    {
110                    String hour   = "";
111                    String minute = "";
112                    if (i < 9)
113                        {
114                        hour = "0" + i;
115                        }
116                    else
117                        {
118                        hour = "" + i;
119                        }
120                    if (j < 9)
121                        {
122                        minute = "0" + j;
123                        }
124                    else
125                        {
126                        minute = "" + j;
127                        }
128                        
129                    AUTOACTIVATION_TIMES.add(hour + ":" + minute);
130                    }
131                }
132            }
133            
134        }
135    
136    /**
137     * Constructs a Conf object.
138     */
139    private Conf()
140        {
141        //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
142        }
143        
144    }
