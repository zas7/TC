017package ch.zh.transferclient.main;
018
019import java.io.File;
020import java.nio.file.Files;
021
022import ch.zh.transferclient.properties.*;
023
024import static java.nio.file.StandardCopyOption.*;
025
026/**
027 * This class is used to delete old receipts.
028 *
029 * @author  Daniel Bierer (Statistical Office of the Canton of Zurich)
030 * @version 2.4
031 */
032public class Treatment_Old_Receipts
033    
034    {
035    
036    /**
037     * Deletes all files within the sedex receipts folder.
038     *
039     * @param  properties Properties to be used.
040     * @throws Exception  The treatment of old properties failed.
041     */
042    protected static void execute(Properties properties) throws Exception
043        {
044        
045        String       sedex_dir_receipts = properties.get_sedex_dir_receipts();
046        
047        final File   dir                = new File(sedex_dir_receipts);
048        
049        final File[] files              = dir.listFiles();
050        
051        for (int i = 0; i < files.length; i++)
052            {
053            
054            Files.move(files[i].toPath(), new File("archive/receipts/"
055                    + files[i].getName()).toPath(), REPLACE_EXISTING);
056            
057            Logger.info("OLD RECEIPT MOVED TO archive/receipts: " + files[i].getName());
058            
059            }
060        
061        //Nachfolgend werden Files ins Archiv verschoben, 
062        //die aufgrund von ausserordentlichen Unterbruechen 
063        //im Stage-Folder zurueckgeblieben sind.
064        File dir_stage = new File("stage/");
065        final File[] files_stage = dir_stage.listFiles();
066        
067        for (int i = 0; i < files_stage.length; i++)
068            {
069            
070            Files.move(files_stage[i].toPath(), new File("archive/stage/"
071                    + files_stage[i].getName()).toPath(), REPLACE_EXISTING);
072            
073            Logger.info("UNPROCESSED FILE MOVED FROM stage TO archive/stage: " + files[i].getName());
074            
075            }
076            
077        }
078        
079    }
