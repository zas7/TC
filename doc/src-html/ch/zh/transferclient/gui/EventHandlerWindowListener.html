017package ch.zh.transferclient.gui;
018
019import java.awt.event.*;
020
021import ch.zh.transferclient.controller.*;
022import ch.zh.transferclient.main.Exit;
023import ch.zh.transferclient.properties.*;
024
025/**
026 * This class is used to handle the window closing event.
027 *
028 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
029 * @version 2.4
030 */
031public class EventHandlerWindowListener implements WindowListener
032    
033    {
034    
035    /** The properties to be used. */
036    private final Properties properties;
037    
038    /** The controller to be used. */
039    private final Controller controller;
040    
041    /**
042     * Constructs the EventHandlerWindowListener
043     * 
044     * @param properties The properties to be used.
045     * @param controller The controller to be used.
046     */
047    protected EventHandlerWindowListener(Properties properties, Controller controller)
048        {
049        this.properties = properties;
050        this.controller = controller;
051        }
052        
053    // WindowListener
054    @Override
055    public void windowClosing(WindowEvent e)
056        {
057        Exit.execute(this.properties, this.controller);
058        }
059        
060    @Override
061    public void windowClosed(WindowEvent e)
062        {
063        }
064        
065    @Override
066    public void windowActivated(WindowEvent e)
067        {
068        }
069        
070    @Override
071    public void windowDeactivated(WindowEvent e)
072        {
073        }
074        
075    @Override
076    public void windowDeiconified(WindowEvent e)
077        {
078        }
079        
080    @Override
081    public void windowIconified(WindowEvent e)
082        {
083        }
084        
085    @Override
086    public void windowOpened(WindowEvent e)
087        {
088        }
089        
090    }
