017package ch.zh.transferclient.gui;
018
019import java.awt.Color;
020import java.awt.Component;
021
022import javax.swing.*;
023import javax.swing.table.*;
024
025import ch.zh.transferclient.main.*;
026import ch.zh.transferclient.properties.*;
027
028/**
029 * This class is used to produce a protocol table.
030 *
031 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
032 * @version 2.4
033 */
034public class FactoryTableProtocol_Detail
035    
036    {
037    
038    /**
039     * Constructs a FactoryTableProtocol_Detail object.
040     */
041    private FactoryTableProtocol_Detail()
042        {
043        //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
044        }
045    
046    
047    /**
048     * Produces the detail protocol table.
049     * 
050     * @param  properties The properties to be used.
051     * @return            Protocol table.
052     */
053    protected static JTable get_table(Properties properties)
054        
055        {
056        
057        DefaultTableModel model = get_model();
058        
059        JTable            table = new JTable(model);
060        
061        table.setColumnSelectionAllowed(false);
062        table.setRowSelectionAllowed(false);
063        
064        // Spaltenbreite der Report-ID Spalte festlegen
065        table.getColumnModel().getColumn(0).setMinWidth(50);
066        table.getColumnModel().getColumn(0).setMaxWidth(50);
067        
068        // Setzen der Selektions-Eigenschaften
069        table.setColumnSelectionAllowed(false);
070        table.setRowSelectionAllowed(false);
071        table.setCellSelectionEnabled(false);
072        
073        // Setzen des ZellenRenderers
074        table.setDefaultRenderer(Object.class, new DefaultTableCellRenderer()
075            {
076            
077            /** A unique serial version identifier. */
078            private static final long serialVersionUID = 1L;
079            
080            @Override
081            public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column)
082                {
083                
084                final Component c                   = super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
085                
086                int             number_of_receivers = properties.get_sedex_recipient_ids().size();
087                Integer         factor              = row / number_of_receivers;
088                if ((factor % 2) == 0)
089                    {
090                    // c.setBackground(Color.WHITE);
091                    c.setBackground(new Color(250, 250, 250)); // Ganz Helles Hellgrau
092                    }
093                else
094                    {
095                    // c.setBackground(Color.LIGHT_GRAY);
096                    c.setBackground(new Color(210, 210, 210)); // Helles Hellgrau
097                    }
098                    
099                if (table.getValueAt(row, 5).toString().contains("successfully"))
100                    {
101                    c.setForeground(Conf.GREEN);
102                    // c.setFont(c.getFont().deriveFont(java.awt.Font.BOLD));
103                    }
104                else
105                    {
106                    c.setForeground(Color.BLACK);
107                    // c.setFont(c.getFont().deriveFont(java.awt.Font.PLAIN));
108                    }
109                    
110                if (column == 0)
111                    {
112                    setHorizontalAlignment(SwingConstants.CENTER);
113                    }
114                else
115                    {
116                    setHorizontalAlignment(SwingConstants.LEFT);
117                    }
118                    
119                return c;
120                }
121            });
122        
123        table.setFocusable(false);
124        table.setRowSelectionAllowed(false);
125        
126        return table;
127        
128        }
129        
130    /**
131     * Produces the table model.
132     * 
133     * @return The table model.
134     */
135    private static DefaultTableModel get_model()
136        {
137        
138        DefaultTableModel model = new DefaultTableModel()
139            {
140            private static final long serialVersionUID = 1L;
141            
142            @Override
143            public boolean isCellEditable(int row, int column)
144                {
145                return false;
146                }
147                
148            };
149        
150        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_0"));
151        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_1"));
152        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_2"));
153        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_3"));
154        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_4"));
155        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_5"));
156        
157        return model;
158        
159        }
160        
161    }
