017package ch.zh.transferclient.gui;
018
019import java.awt.Component;
020
021import javax.swing.*;
022import javax.swing.border.*;
023
024import ch.zh.transferclient.main.*;
025
026/**
027 * This class is used to show a dialog which informs the user that the transfer-client will be deactivated.
028 *
029 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
030 * @version 2.4
031 */
032public class DialogWaitingForDeactivation extends JDialog
033    
034    {
035    
036    /** A unique serial version identifier. */
037    private static final long  serialVersionUID = 1L;
038    
039    /** The width of the dialog. */
040    protected static final int WIDTH            = 320;
041    
042    /** The height of the dialog. */
043    protected static final int HEIGHT           = 130;
044    
045    /** The OK button. */
046    private final Gui          gui;
047    
048    /**
049     * Makes the dialog visible.
050     *
051     */
052    public void setvisible()
053        {
054        this.gui.setEnabled(false);
055        this.setVisible(true);
056        }
057        
058    /**
059     * Makes the dialog unvisible.
060     * 
061     */
062    public void setunvisible()
063        {
064        this.gui.setEnabled(true);
065        this.setVisible(false);
066        }
067        
068    /**
069     * Constructs a DialogWaitingForDeactivation object.
070     * 
071     * @param parent The parent frame to be used.
072     */
073    public DialogWaitingForDeactivation(JFrame parent)
074        
075        {
076        
077        // Grundparameter
078        super(parent, Labels.get("DIALOG_WAITINGFORDEACTIVATION_TITLE"));
079        this.gui = (Gui) parent;
080        this.setSize(WIDTH, HEIGHT);
081        this.setIconImage(Conf.IMAGE_FOR_ICON);
082        this.setResizable(false);
083        
084        // Exit-Operation definieren
085        this.setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
086        
087        // Panel
088        JPanel      panel       = new JPanel();
089        EmptyBorder emptyborder = new EmptyBorder(20, 20, 20, 20);
090        panel.setBorder(emptyborder);
091        panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
092        
093        // Label
094        JLabel label_hinweis = new JLabel(Labels.get("DIALOG_WAITINGFORDEACTIVATION_TEXT"));
095        label_hinweis.setAlignmentX(Component.LEFT_ALIGNMENT);
096        panel.add(label_hinweis);
097        
098        panel.add(new JPanel());
099        
100        setContentPane(panel);
101        
102        }
103        
104    }
