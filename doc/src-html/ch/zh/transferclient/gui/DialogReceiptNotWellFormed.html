017package ch.zh.transferclient.gui;
018
019import java.awt.*;
020import java.awt.event.*;
021
022import javax.swing.*;
023import javax.swing.border.*;
024
025import ch.zh.transferclient.controller.*;
026import ch.zh.transferclient.main.*;
027import ch.zh.transferclient.properties.*;
028
029/**
030 * This class is used to show a dialog which informs the user that a received sedex receipt file is not well formed.
031 *
032 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
033 * @version 2.4
034 */
035public class DialogReceiptNotWellFormed extends JDialog implements MouseListener, WindowListener
036    
037    {
038    
039    /** A unique serial version identifier. */
040    private static final long  serialVersionUID = 1L;
041    
042    /** The width of the dialog. */
043    protected static final int WIDTH            = 400;
044    
045    /** The height of the dialog. */
046    protected static final int HEIGHT           = 130;
047    
048    /** The graphical user interface object to be used. */
049    private final Gui          gui;
050    
051    /** The properties to be used. */
052    private final Properties   properties;
053    
054    /** The controller to be used. */
055    private final Controller   controller;
056    
057    /** The button. */
058    private final JButton      button           = new JButton(Labels.get("DIALOG_RECEIPTNOTWELLFORMED_BUTTON"));
059    
060    // MouseListener-Methoden
061    @Override
062    public void mouseClicked(MouseEvent e)
063        {
064        Exit.execute(this.properties, this.controller);
065        }
066        
067    @Override
068    public void mouseReleased(MouseEvent e)
069        {
070        }
071        
072    @Override
073    public void mousePressed(MouseEvent e)
074        {
075        }
076        
077    @Override
078    public void mouseExited(MouseEvent e)
079        {
080        }
081        
082    @Override
083    public void mouseEntered(MouseEvent e)
084        {
085        }
086        
087    // WindowListener-Methoden
088    @Override
089    public void windowClosing(WindowEvent e)
090        {
091        Exit.execute(this.properties, this.controller);
092        }
093        
094    @Override
095    public void windowClosed(WindowEvent e)
096        {
097        }
098        
099    @Override
100    public void windowActivated(WindowEvent e)
101        {
102        }
103        
104    @Override
105    public void windowDeactivated(WindowEvent e)
106        {
107        }
108        
109    @Override
110    public void windowDeiconified(WindowEvent e)
111        {
112        }
113        
114    @Override
115    public void windowIconified(WindowEvent e)
116        {
117        }
118        
119    @Override
120    public void windowOpened(WindowEvent e)
121        {
122        }
123        
124    /**
125     * Makes the dialog visible.
126     */
127    public void setvisible()
128        {
129        this.gui.setEnabled(false);
130        this.setVisible(true);
131        }
132        
133    /**
134     * Constructs a DialogReceiptNotWellFormed object.
135     * 
136     * @param parent     The parent frame to be used.
137     * @param properties The properties to be used.
138     * @param controller The controller to be used.
139     */
140    public DialogReceiptNotWellFormed(JFrame parent, Properties properties, Controller controller)
141        
142        {
143        
144        // Grundparameter
145        super(parent, Labels.get("DIALOG_RECEIPTNOTWELLFORMED_TITLE"));
146        this.gui        = (Gui) parent;
147        this.properties = properties;
148        this.controller = controller;
149        this.setSize(WIDTH, HEIGHT);
150        this.setIconImage(Conf.IMAGE_FOR_ICON);
151        this.setResizable(false);
152        
153        // EventHandlers
154        this.button.addMouseListener(this);
155        this.addWindowListener(this);
156        
157        // Exit-Operation definieren
158        this.setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
159        
160        // OK-Knopf
161        this.button.setAlignmentX(Component.CENTER_ALIGNMENT);
162        
163        // Border
164        EmptyBorder emptyborder = new EmptyBorder(20, 20, 20, 20);
165        
166        // JPanel/Layout
167        JPanel      panel       = new JPanel();
168        panel.setBorder(emptyborder);
169        panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
170        
171        // Hinweis-Text
172        JPanel jp_hinweis = new JPanel();
173        jp_hinweis.setLayout(new BoxLayout(jp_hinweis, BoxLayout.Y_AXIS));
174        JLabel label_hinweis_1 = new JLabel(Labels.get("DIALOG_RECEIPTNOTWELLFORMED_TEXT_1"));
175        JLabel label_hinweis_2 = new JLabel(Labels.get("DIALOG_RECEIPTNOTWELLFORMED_TEXT_2"));
176        label_hinweis_1.setAlignmentX(Component.LEFT_ALIGNMENT);
177        label_hinweis_2.setAlignmentX(Component.LEFT_ALIGNMENT);
178        jp_hinweis.add(label_hinweis_1);
179        jp_hinweis.add(label_hinweis_2);
180        jp_hinweis.setAlignmentX(Component.CENTER_ALIGNMENT);
181        panel.add(jp_hinweis);
182        
183        panel.add(new JPanel());
184        
185        // Knopf
186        panel.add(this.button);
187        
188        setContentPane(panel);
189        
190        }
191        
192    }
