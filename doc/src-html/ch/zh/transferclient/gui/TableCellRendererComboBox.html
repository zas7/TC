036    /**
037     * Constructs a TableCellRendererComboBox object.
038     * 
039     * @param items Items to be used.
040     */
041    public TableCellRendererComboBox(String[] items)
042        {
043        super(items);
044        super.setFont(new Font("Sanserif", Font.PLAIN, 12)); // Das durch die ComboBox ausgewaehlte Item soll nicht fett
045                                                             // sein.
046        
047        // Super Variante zur Lösung des Border-Problems

057        // this.setBorder(BorderFactory.createLineBorder(Color.WHITE));
058        
059        }
060        
061    @Override
062    public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column)
063        {
064        if (isSelected)
065            {
066            super.setForeground(table.getSelectionForeground());
067            super.setBackground(table.getSelectionBackground());
068            }
069        else
070            {
071            super.setForeground(table.getForeground());
072            super.setBackground(table.getBackground());
073            }
074        super.setSelectedItem(value);
075        
076        return this;
077        }
078        
079    }
