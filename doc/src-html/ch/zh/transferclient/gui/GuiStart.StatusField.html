017package ch.zh.transferclient.gui;
018
019import java.awt.*;
020
021import javax.swing.*;
022import javax.swing.border.*;
023
024import ch.zh.transferclient.main.*;
025
026/**
027 * This class is used to show an initial frame, informing that the transfer-client is starting up.
028 *
029 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
030 * @version 2.4
031 */
032public class GuiStart extends JFrame
033    
034    {
035    
036    /**
037     * Status fields of the GuiStart that can be used for informing the end user about the current status.
038     */
039    public static enum StatusField {
040        STATUSFIELD1,
041        STATUSFIELD2,
042        STATUSFIELD3
043        }
044    
045    /** A unique serial version identifier. */
046    private static final long serialVersionUID = 1L;
047    
048    /** The width of the dialog. */
049    public static final int   WIDTH            = 400;
050    
051    /** The height of the dialog. */
052    public static final int   HEIGHT           = 180;
053    
054    /** The status field 1. */
055    private final JLabel statusfield1;
056    
057    /** The status field 2. */
058    private final JLabel statusfield2;
059    
060    /** The status field 3. */
061    private final JLabel statusfield3;
062    
063    /**
064     * Updates the status field.
065     * @param field The field to update.
066     * @param text The new text.
067     */
068    public void update_statusfield(StatusField field,String text)
069        {
070        JLabel targetfield;
071        if (field==StatusField.STATUSFIELD1)        {targetfield=this.statusfield1;}
072        else if (field==StatusField.STATUSFIELD2)   {targetfield=this.statusfield2;}
073        else                                        {targetfield=this.statusfield3;}
074        
075        targetfield.setText(text);
076        targetfield.repaint();
077        }
078    
079    /**
080     * Constructs a GuiStart object.
081     */
082    public GuiStart()
083        
084        {
085        
086        super("Transfer-Client " + Conf.VERSION);
087        
088        // -----------------------------------------//
089        // Icon, Frame-Zentrierung, Exit-Operation //
090        // -----------------------------------------//
091        
092        this.setIconImage(Conf.IMAGE_FOR_ICON);
093        
094        // Frame-Zentrierung
095        this.setSize(WIDTH, HEIGHT);
096        Dimension frameSize  = getSize();
097        Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
098        int       x          = (screenSize.width - frameSize.width) / 2;
099        int       y          = (screenSize.height - frameSize.height) / 2;
100        this.setLocation(x, y);
101        
102        // Exit-Operation definieren
103        this.setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
104        
105        JPanel panel = new JPanel();
106        panel.setBorder(new EmptyBorder(20, 20, 20, 20));
107        panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
108        
109        // --------------//
110        // Inneres Panel //
111        // --------------//
112        JPanel innerpanel = new JPanel();
113        FlowLayout flowlayout = new FlowLayout(FlowLayout.LEFT,5,5);
114        innerpanel.setLayout(flowlayout);
115        
116        // ---------------------------------------------//
117        // Inneres Panel: Linke Seite: Bild hinzufuegen //
118        // ---------------------------------------------//
119        JLabel bild = new JLabel(Conf.IMAGE_ICON_LOGO);
120        innerpanel.add(bild);
121        
122        
123        // ----------------------------------------------//
124        // Inneres Panel: Rechte Seite: Text hinzufuegen //
125        // ----------------------------------------------//
126        //Panel fuer Text1 und Text2
127        JPanel textpanel = new JPanel();
128        GridLayout gridlayout = new GridLayout(4,1); //Es sind 4 Texte untereinander.
129        textpanel.setLayout(gridlayout);
130        
131        //Titelzeile
132        JLabel titel = new JLabel("Initialization");
133        Font font_titel = new Font("Sans Serif", Font.BOLD, 16);
134        titel.setFont(font_titel);
135        textpanel.add(titel);
136        
137        //Erste Textzeile: Statusfeld 1
138        Font font_status = new Font("Sans Serif", Font.PLAIN, 12);
139        this.statusfield1 = new JLabel("");        
140        this.statusfield1.setFont(font_status);
141        textpanel.add(this.statusfield1);
142        
143        //Zweite Textzeile: Statusfeld 2
144        this.statusfield2 = new JLabel("");
145        this.statusfield2.setFont(font_status);
146        textpanel.add(this.statusfield2);
147        
148        //Dritte Textzeile: Statusfeld 3
149        this.statusfield3 = new JLabel("");
150        this.statusfield3.setFont(font_status);
151        textpanel.add(this.statusfield3);
152        
153        innerpanel.add(textpanel);
154        
155        // ----------------------------------------------//
156        // Inneres Panel dem aeusseren Panel hinzufuegen //
157        // ----------------------------------------------//
158        panel.add(innerpanel);
159        
160        // Zwischenraum
161        JLabel empty = new JLabel("");
162        panel.add(empty);
163        
164        // ------------//
165        // GUI-Objekt //
166        // ------------//
167        this.setContentPane(panel);
168        this.setResizable(false);
169        this.setVisible(true);
170        
171        }
172        
173    }
