017package ch.zh.transferclient.gui;
018
019import java.awt.Color;
020import java.awt.Component;
021
022import javax.swing.*;
023import javax.swing.table.*;
024
025import ch.zh.transferclient.main.*;
026import ch.zh.transferclient.properties.*;
027
028/**
029 * This class is used to produce a protocol table.
030 *
031 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
032 * @version 2.4
033 */
034public class FactoryTableProtocol_Overview
035    
036    {
037    
038    /**
039     * Constructs a FactoryTableProtocol_Overview object.
040     */
041    private FactoryTableProtocol_Overview()
042        {
043        //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
044        }
045    
046    /**
047     * Produces the overview protocol table.
048     * 
049     * @param  properties The properties to be used.
050     * @return            Protocol table.
051     */
052    protected static JTable get_table(Properties properties)
053        
054        {
055        
056        DefaultTableModel model = get_model();
057        
058        JTable            table = new JTable(model);
059        
060        table.setColumnSelectionAllowed(false);
061        table.setRowSelectionAllowed(false);
062        
063        // Spaltenbreite der Report-ID Spalte festlegen
064        table.getColumnModel().getColumn(0).setMinWidth(50);
065        table.getColumnModel().getColumn(0).setMaxWidth(50);
066        
067        // Setzen der Selektions-Eigenschaften
068        table.setColumnSelectionAllowed(false);
069        table.setRowSelectionAllowed(false);
070        table.setCellSelectionEnabled(false);
071        
072        // Setzen des ZellenRenderers
073        table.setDefaultRenderer(Object.class, new DefaultTableCellRenderer()
074            {
075            
076            /** A unique serial version identifier. */
077            private static final long serialVersionUID = 1L;
078            
079            @Override
080            public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column)
081                {
082                
083                final Component c                    = super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
084                
085                final Integer   NUMBER_OF_RECIPIENTS = properties.get_sedex_recipient_ids().size();
086                final String    KEY_TEXT             = NUMBER_OF_RECIPIENTS + " of " + NUMBER_OF_RECIPIENTS;
087                
088                if (table.getValueAt(row, 3).toString().contains(KEY_TEXT))
089                    {
090                    c.setForeground(Conf.GREEN);
091                    // c.setFont(c.getFont().deriveFont(java.awt.Font.BOLD));
092                    }
093                else
094                    {
095                    c.setForeground(Color.BLACK);
096                    // c.setFont(c.getFont().deriveFont(java.awt.Font.PLAIN));
097                    }
098                    
099                if (column == 0)
100                    {
101                    setHorizontalAlignment(SwingConstants.CENTER);
102                    }
103                else
104                    {
105                    setHorizontalAlignment(SwingConstants.LEFT);
106                    }
107                    
108                return c;
109                }
110            });
111        
112        table.setFocusable(false);
113        table.setRowSelectionAllowed(false);
114        
115        return table;
116        
117        }
118        
119    /**
120     * Produces the table model.
121     * 
122     * @return The table model.
123     */
124    private static DefaultTableModel get_model()
125        {
126        
127        DefaultTableModel model = new DefaultTableModel()
128            {
129            private static final long serialVersionUID = 1L;
130            
131            @Override
132            public boolean isCellEditable(int row, int column)
133                {
134                return false;
135                }
136                
137            };
138        
139        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_0")); // Heure
140        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_1")); // Données diffusées
141        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_4")); // ID Message
142        model.addColumn(Labels.get("FACTORYTABLEPROTOCOL_COLUMN_5")); // Status
143        
144        return model;
145        
146        }
147        
148    }
