464    /**
465     * 
466     * A factory method is used to offer GUI objects to the client (please note that the constructor has been declared
467     * as private).
468     * 
469     * What is a factory method? See Goetz (2006): Java Concurrency in Practice, Chapter 3.2.1, see page 42 for details.
470     * 
471     * Why are the GUI objects not offered to the client the normal way, i.e. by a public constructor?
472     * 
473     * If a GUI object would be offered directly by the constructor to the client, it would be necessary to register the
474     * listeners inside the constructor. However, in this case, the "this"-reference would escape during construction
475     * because the listeners need a reference to the GUI object: Example code of a constructor with escaping
476     * this-reference:
477     * 
478     * <pre>
479     * Controller controller = new Controller(this, properties); // this-reference is escaping
480     * this.button_activation.addMouseListener(controller); // registering listener
481     * this.button_deactivation.addMouseListener(controller); // registering listener
482     * this.button_autoactivation.addMouseListener(controller); // registering listener
483     * this.textfield_folder.addMouseListener(controller); // registering listener
484     * this.button_folder.addMouseListener(controller); // registering listener
485     * </pre>
486     * 
487     * If the this-reference escapes during construction, the object is considered not properly constructed (see page 41
488     * in Goetz (2006)).
489     * 
490     * Therefore, the factory method constructs firstly the GUI object and after construction has finished, it registers
491     * the event listeners.
492     * 
493     * Book References: Goetz (2006): Java Concurrency in Practice, Chapter 3.2.1, see page 42 for details.
494     * 
495     * Internet References: https://stackoverflow.com/questions/3705425/java-reference-escape
496     * https://stackoverflow.com/questions/27453105/call-a-method-after-the-constructor-has-ended
497     * https://stackoverflow.com/questions/13717603/guice-way-to-register-a-listener-after-calling-constructor
498     * 
499     * 
500     * An alternative (very common) approach would be to use inner classes for the event handlers. However, this
501     * approach would have inflated the GUI class too much.
502     *
503     * Book References: Sierra, Bates: Head First Java (A Brain Friendly Guide), Chapter 12, Page 374ff. Horstmann: Java
504     * for Everyone, Chapter 11.6: Using Inner Classes for Listeners
505     *
506     * Internet References: https://docs.oracle.com/javase/tutorial/uiswing/events/generalrules.html#innerClasses
507     * https://stackoverflow.com/questions/1894181/standards-for-using-inner-classes-for-gui
508     * https://stackoverflow.com/questions/15960950/better-way-to-implement-inner-class-with-listeners
509     * http://www.fredosaurus.com/notes-java/GUI/events/inner_class_listener.html
510     * 
511     *
512 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
513 * @version 2.4
514     * 
515     * @param   properties Reference of the properties object to be used.
516     * @return             The manufactured GUI object.
517     * 
518     */
519    public static Gui get_newInstance(Properties properties)
520        {
521        
522        // ---------------------------------------------------------------------------------
523        // SCHRITT 0: KONSTRUKTION DES GUI OBJEKTS
524        // ---------------------------------------------------------------------------------
525        Gui        gui        = new Gui(properties);
526        
527        // ---------------------------------------------------------------------------------
528        // SCHRITT 1: KONSTRUKTION DES CONTROLLERS
529        // ---------------------------------------------------------------------------------
530        // Bemerkungen:
531        // a) Der Controller kann erst nach der Konstruktion des GUI-Objekts
532        // konstruiert werden, da zu dessen Konstruktion eine Referenz
533        // auf das GUI-Objekt benoetigt wird.
534        // b) Der Controller wird als Mouse-Listener fuer gewisse
535        // GUI-Objekte verwendet (vgl. Source Code unten).
536        // c) Der Controller wird zur Konstruktion von Dialogen und weiteren
537        // Listenern benoetigt.
538        // ---------------------------------------------------------------------------------
539        Controller controller = new Controller(gui, properties);
540        
541        // ---------------------------------------------------------------------------------
542        // SCHRITT 2: REGISTRIERUNG DES CONTROLLERS ALS MOUSE_LISTENER FUER GUI-OBJEKTE
543        // ---------------------------------------------------------------------------------
544        gui.get_button_activation().addMouseListener(controller);
545        gui.get_button_deactivation().addMouseListener(controller);
546        gui.get_button_autoactivation().addMouseListener(controller);
547        gui.get_textfield_folder().addMouseListener(controller);
548        gui.get_button_folder().addMouseListener(controller);
549        controller.deactivation();
550        
551        // ---------------------------------------------------------------------------------
552        // SCHRITT 3: KONSTRUKTION DER DIAGLOGE FUER DAS GUI-OBJEKT
553        // ---------------------------------------------------------------------------------
554        // Die Dialoge werden nicht im Rahmen der Konstruktion des GUI-Objekts konstruiert,
555        // da sonst eine this-reference fluechten koennte, wenn das GUI-Objekt noch
556        // nicht fertig konstruiert ist.
557        // ---------------------------------------------------------------------------------
558        
559        DialogDir                    dialog_dir                    = new DialogDir(gui);
560        DialogFileProcessingError    dialog_fileprocessingerror    = new DialogFileProcessingError(gui);
561        DialogReceiptNotWellFormed   dialog_receiptnotwellformed   = new DialogReceiptNotWellFormed(gui, properties, controller);
562        DialogSupport                dialog_support                = new DialogSupport(gui, false);
563        DialogWaitingForDeactivation dialog_waitingfordeactivation = new DialogWaitingForDeactivation(gui);
564        dialog_dir.setVisible(false);
565        dialog_fileprocessingerror.setVisible(false);
566        dialog_receiptnotwellformed.setVisible(false);
567        dialog_support.setVisible(false);
568        dialog_waitingfordeactivation.setVisible(false);
569        gui.set_dialog_dir(dialog_dir);
570        gui.set_dialog_fileprocessingerror(dialog_fileprocessingerror);
571        gui.set_dialog_receiptnotwellformed(dialog_receiptnotwellformed);
572        gui.set_dialog_support(dialog_support);
573        gui.set_dialog_waitingfordeactivation(dialog_waitingfordeactivation);
574        
575        // ---------------------------------------------------------------------------------
576        // SCHRITT 4: KONSTRUKTION VON WEITERERN LISTERNERN
577        // ---------------------------------------------------------------------------------
578        // Die Konstruktion der folgenden weiteren EventListener wird nicht im Rahmen
579        // der Konstruktion des GUI-Objekts vollzogen, da zu deren Konstruktion eine
580        // Referenz auf das GUI-Objekt benoetigt wird.
581        // ---------------------------------------------------------------------------------
582        
583        // ActionListener
584        // Steuerung des Balkenmenüs
585        EventHandlerActionListener actionlistener = new EventHandlerActionListener(gui, properties, controller);
586        gui.get_menuitem0().addActionListener(actionlistener);
587        gui.get_menuitem1().addActionListener(actionlistener);
588        
589        // ComponentListener
590        // Beim Verschieben des Hauptfensters werden die Unterfenster auch
591        // mitverschoben.
592        // Beim Verschieben von Unterfenstern wird das Hauptfesnter auch mitverschoben.
593        EventHandlerComponentListener componentlistener = new EventHandlerComponentListener(gui);
594        gui.addComponentListener(componentlistener);
595        gui.get_dialog_dir().addComponentListener(componentlistener);
596        gui.get_dialog_fileprocessingerror().addComponentListener(componentlistener);
597        gui.get_dialog_support().addComponentListener(componentlistener);
598        gui.get_dialog_receiptnotwellformed().addComponentListener(componentlistener);
599        gui.get_dialog_waitingfordeactivation().addComponentListener(componentlistener);
600        
601        // ItemListener
602        // Wir brauchen fuer die ComoBox einen ItemListener und keinen ActionListener.
603        // https://stackoverflow.com/questions/32752072/jcombobox-and-itemlistener-actionlistener
604        EventHandlerItemListener itemlistener = new EventHandlerItemListener(gui, properties);
605        gui.get_combobox_autoactivation().addItemListener(itemlistener);
606        
607        // ---------------------------------------------------------------------------------
608        // SCHRITT 5: KONSTRUKTION VON WEITERERN LISTERNERN
609        // ---------------------------------------------------------------------------------
610        // Zur Konstruktion des folgenden Listeners wird zwar keine this-Referenz
611        // benoetigt, jedoch eine Referenz auf das Controller-Objekt, welches
612        // seinerseits
613        // eine Referenz auf das GUI-Objekt benoetigt. Aus diesem Grund muss auch der
614        // folgende Listener nach der Konstruktion des GUI-Objekts konstruiert werden.
615        //
616        //
617        // Festlegung derjenigen EventHandler, die keine
618        // this-reference zur Konstruktion benoetigen
619        // ----------------------------------------------------------------------------------
620        // WindowListener
621        // Festlegung der Exit-Aktion
622        EventHandlerWindowListener windowlistener = new EventHandlerWindowListener(properties, controller);
623        gui.addWindowListener(windowlistener);
624        
625        // -------------------------------------
626        // Returning the properly constructed
627        // GUI object to the client
628        // -------------------------------------
629        return gui;
630        
631        }
