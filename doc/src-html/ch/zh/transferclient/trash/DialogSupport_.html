017package ch.zh.transferclient.trash;
018
019import java.awt.*;
020import java.awt.event.*;
021
022import javax.swing.*;
023import javax.swing.border.*;
024
025import ch.zh.transferclient.main.Conf;
026
027/**
028 * This class is used for constructing a JDialog.
029 *
030 * @author  Daniel Bierer (Statistical Office of the Canton of Zurich)
031 * @version 2.4
032 */
033public class DialogSupport_ extends JDialog implements MouseListener
034    
035    {
036    
037    /** A unique serial version identifier. */
038    private static final long  serialVersionUID = 1L;
039    
040    /** The width of the dialog. */
041    protected static final int WIDTH            = 400;
042    
043    /** The height of the dialog. */
044    protected static final int HEIGHT           = 150;
045    
046    /** A button. */
047    private final JButton      okbutton         = new JButton("OK");
048    
049    @Override
050    public void mouseClicked(MouseEvent e)
051        {
052        
053        if (e.getSource() == this.okbutton)
054            {
055            this.setVisible(false);
056            }
057        }
058        
059    @Override
060    public void mouseReleased(MouseEvent e)
061        {
062        }
063        
064    @Override
065    public void mousePressed(MouseEvent e)
066        {
067        }
068        
069    @Override
070    public void mouseExited(MouseEvent e)
071        {
072        }
073        
074    @Override
075    public void mouseEntered(MouseEvent e)
076        {
077        }
078        
079    /**
080     * Constructs a JDialog object.
081     * 
082     * @param parent The parent frame to be used.
083     */
084    public DialogSupport_(JFrame parent)
085        
086        {
087        
088        // Grundparameter
089        super(parent, "Support");
090        this.setSize(WIDTH, HEIGHT);
091        this.setIconImage(Conf.IMAGE_FOR_ICON);
092        this.setResizable(false);
093        
094        //
095        this.okbutton.addMouseListener(this);
096        
097        // OK-Button
098        this.okbutton.setAlignmentX(Component.CENTER_ALIGNMENT);
099        
100        // Border
101        EmptyBorder emptyborder = new EmptyBorder(20, 20, 20, 20);
102        
103        // JPanel/Layout
104        JPanel      panel       = new JPanel();
105        panel.setBorder(emptyborder);
106        panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
107        
108        // Jpanel_gridbag/Layout
109        JPanel panel_gridbag = new JPanel();
110        panel_gridbag.setLayout(new GridBagLayout());
111        panel_gridbag.setAlignmentX(Component.CENTER_ALIGNMENT);
112        
113        // Constraints
114        GridBagConstraints c = new GridBagConstraints();
115        
116        c.insets = new Insets(5, 5, 5, 5);
117        
118        // Zeile 1 / Spalte 1
119        c.anchor = GridBagConstraints.WEST;
120        panel_gridbag.add(new JLabel("SEDEX Support:"), c);
121        
122        // Zeile 1 / Spalte 2
123        c.anchor    = GridBagConstraints.WEST;
124        c.gridwidth = GridBagConstraints.REMAINDER;
125        panel_gridbag.add(new JLabel("047 333 22 11"), c);
126        
127        // Zeile 2 / Spalte 1
128        c.anchor    = GridBagConstraints.WEST;
129        c.gridwidth = GridBagConstraints.RELATIVE;
130        panel_gridbag.add(new JLabel("Transfer-Client:"), c);
131        
132        // Zeile 2 / Spalte 2
133        c.anchor    = GridBagConstraints.WEST;
134        c.gridwidth = GridBagConstraints.REMAINDER;
135        panel_gridbag.add(new JLabel("044 777 33 11"), c);
136        
137        // GridBag
138        panel.add(panel_gridbag);
139        
140        panel.add(new JPanel());
141        
142        // Knopf
143        panel.add(this.okbutton);
144        
145        setContentPane(panel);
146        
147        }
148        
149    }
