017package ch.zh.transferclient.annotationservices;
018
019import java.io.*;
020import java.nio.charset.StandardCharsets;
021import java.nio.file.*;
022import java.util.*;
023
024/**
025 * The class is used to update systematically the author and version of all source files.
026 * 
027 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
028 * @version 2.4
029 *
030 */
031public class AnnotationsUpdate
032    {
033    
034    /**
035     * Constructs an AnnotationsUpdate object.
036     */
037    private AnnotationsUpdate()
038        {
039        //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
040        }
041    
042    
043    /**
044     * Updates the author and version of all source code files.
045     * 
046     * @param rootfolder The root folder to be used.
047     * 
048     */
049    protected static void update(String rootfolder)
050        {
051        
052        try
053            {
054            Iterator<Path> it = Files.walk(Paths.get(rootfolder)).filter(p -> (p.toString().endsWith(".java") || p.toString().endsWith("package.html") )).iterator();
055            
056            while (it.hasNext())
057                {
058                Path              path  = it.next();
059                File              file  = path.toFile();
060                
061                // ------------------------------//
062                // Einlesen des bisherigen Files //
063                // ------------------------------//
064                BufferedReader    br    = new BufferedReader(new InputStreamReader(new FileInputStream(file), "UTF-8"));
065                ArrayList<String> lines = new ArrayList<String>();
066                
067                String            line  = br.readLine();
068                while ((line != null))
069                    {
070                    if (file.getName().endsWith(".java"))
071                        {
072                        if ((line.contains("* Copyright ")) && ((line.contains("Statistisches Amt des Kantons Zürich"))))
073                            {
074                            line = AnnotationsMain.LICENCEINFO_1FSTLINE;
075                            }
076                        if (line.contains("@author"))
077                            {
078                            line = AnnotationsMain.AUTHOR;
079                            }
080                        if (line.contains("@version"))
081                            {
082                            line = AnnotationsMain.VERSION;
083                            }
084                        }
085                    else if (file.getName().equals("package.html"))
086                        {
087                        if (line.contains("@author"))
088                            {
089                            line = AnnotationsMain.PACKAGES_HTML_AUTHOR;
090                            }
091                        if (line.contains("@version"))
092                            {
093                            line = AnnotationsMain.PACKAGES_HTML_VERSION;
094                            }
095                        }
096                        
097                    
098                    lines.add(line);
099                    
100                    line = br.readLine();
101                    }
102                    
103                br.close();
104                
105                // ------------------------------------//
106                // Ueberschreiben des bisherigen Files //
107                // ------------------------------------//
108                BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(file, false), StandardCharsets.UTF_8));
109                
110                for (int i = 0; i < lines.size(); i++)
111                    {
112                    bw.write(lines.get(i));
113                    bw.newLine();
114                    }
115                    
116                bw.flush();
117                bw.close();
118                
119                System.out.println("FILE UPDATED: " + file.getAbsolutePath());
120                
121                }
122                
123            }
124        catch (Exception e)
125            {
126            e.printStackTrace();
127            }
128            
129        }
130    
131    }
