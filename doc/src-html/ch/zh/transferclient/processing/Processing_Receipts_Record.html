001package ch.zh.transferclient.processing;
002
003/**
004 * This class is used to construct receipts.
005 * 
006 * @author  Daniel Bierer (Statistisches Amt des Kantons ZÃ¼rich)
007 * @version 2.4
008 */
009public class Processing_Receipts_Record
010    {
011    
012    /**
013     * Sedex Message ID.
014     */
015    private final String sedex_message_id;
016    
017    /**
018     * Trimmed Sedex Message ID, i.e. the Sedex Message ID without the recipient information (for example without
019     * "-E0").
020     */
021    private final String sedex_message_id_trimmed;
022    
023    /**
024     * Status Info of the transmission.
025     */
026    private final String status_info;
027    
028    /**
029     * Constructs a receipt.
030     * @param sedex_message_id Sedex Message ID.
031     * @param status_info      Status Info of the transmission
032     */
033    protected Processing_Receipts_Record(String sedex_message_id, String status_info)
034        {
035        this.sedex_message_id         = sedex_message_id;
036        this.sedex_message_id_trimmed = sedex_message_id.substring(0, sedex_message_id.indexOf("-E"));
037        this.status_info              = status_info;
038        
039        }
040        
041    /**
042     * @return The Sedex Message ID.
043     */
044    protected String get_sedex_message_id()
045        {
046        return this.sedex_message_id;
047        }
048        
049    /**
050     * @return The trimmed Sedex Message ID, i.e. the Sedex Message ID without the recipient information (for example
051     *         without "-E0").
052     */
053    protected String get_sedex_message_id_trimmed()
054        {
055        return this.sedex_message_id_trimmed;
056        }
057        
058    /**
059     * @return The status info of the transmission.
060     */
061    protected String get_status_info()
062        {
063        return this.status_info;
064        }
065        
066    }
